\chapter{Đánh giá Hiệu năng Hệ thống (Performance Benchmarking)}
\label{ch:benchmarking}

\section{Thiết lập Môi trường Thử nghiệm Hiệu năng}
Để kiểm tra tính bền vững và khả năng mở rộng của hệ thống khi khối lượng dữ liệu tiệm cận quy mô Big Data, chúng tôi tiến hành đo đạc hiệu năng (Benchmarking) thông qua script tự động \texttt{src/evaluation/benchmark\_multiscale.py}.
\begin{itemize}
    \item \textbf{Môi trường phần cứng:} CPU Intel Core i7-11800H @ 2.30GHz (8 cores, 16 threads), 16GB RAM DDR4, SSD NVMe 512GB. Elasticsearch chạy trên Docker container WSL2 Ubuntu 22.04 LTS.
    \item \textbf{Mục tiêu đo đạc:}
      \begin{itemize}
          \item \textbf{Thời gian phản hồi (Latency):} Đo lường thời gian xử lý truy vấn ở các mức phân vị P50 và P95.
          \item \textbf{Dung lượng chỉ mục (Index Size):} Dung lượng đĩa cứng vật lý tiêu tốn bởi Elasticsearch để lưu trữ tài liệu và các cấu trúc chỉ mục tương ứng.
      \end{itemize}
    \item \textbf{Quy mô dữ liệu (Scales):} Thực hiện đo đạc song song trên ba quy mô tập dữ liệu tăng dần:
      \begin{itemize}
          \item \textbf{100K Scale:} Tập dữ liệu thử nghiệm ban đầu (100,000 bài báo).
          \item \textbf{500K Scale:} Tập dữ liệu quy mô trung bình (500,000 bài báo).
          \item \textbf{1.2M Scale (Full):} Toàn bộ tập dữ liệu lọc được từ arXiv (1,203,108 bài báo).
      \end{itemize}
\end{itemize}

\subsection{Phân tích và Biện luận kết quả Latency}
\begin{itemize}
    \item \textbf{Hiện tượng Cold Start và Page Cache Warming:} Trong quá trình đo đạc thực tế ở quy mô 1.2M, chúng tôi nhận thấy độ trễ của kNN độc lập tăng vọt (P50 đạt 191.3ms, P95 lên tới 1205.8ms) trong khi Hybrid Search lại có độ trễ thấp hơn đáng kể (P50 = 99.0ms, P95 = 156.4ms). Điều này phản ánh rõ cơ chế hoạt động của Lucene và hệ điều hành: Ở lượt truy vấn kNN đầu tiên, đồ thị HNSW lớn phải nạp từ ổ đĩa SSD vào OS Page Cache dẫn đến nghẽn I/O (Cold Start). Khi cache đã được làm ấm (Warmed Cache), các lượt truy vấn tiếp theo (bao gồm Hybrid Search) truy xuất trực tiếp trên RAM và đạt hiệu năng rất cao.
    \item \textbf{BM25 Search:} Vẫn duy trì ưu thế vượt trội về mặt tốc độ ở quy mô nhỏ (P50 = 15.3ms ở 100k). Khi nâng lên 1.2M, P50 tăng lên 70.6ms và P95 đạt 152.6ms do khối lượng tài liệu trong mỗi segment tăng lên, khiến quá trình tính điểm TF-IDF và duyệt posting list tiêu tốn nhiều CPU hơn.
    \item \textbf{Hybrid Search (RRF):} Cho thấy khả năng kiểm soát độ trễ rất tốt sau khi bộ đệm hệ thống đã được tối ưu. Với độ trễ P50 là 99.0ms ở quy mô lớn nhất 1.2M, Hybrid Search hoàn toàn đáp ứng được tiêu chuẩn thời gian thực (dưới 100ms) của các hệ thống tìm kiếm công nghiệp.
\end{itemize}

\section{Đánh giá Dung lượng Lưu trữ (Index Storage Size)}
Dung lượng vật lý tiêu tốn trên đĩa cứng tương ứng với từng quy mô dữ liệu được ghi nhận thực tế trong Bảng \ref{tab:index-size}:

\textbf{Ghi chú về dữ liệu gốc:} File dữ liệu thô tải từ Kaggle (\texttt{arxiv-metadata-oai-snapshot.json}) có dung lượng khoảng \textbf{3.6 GB}, chứa toàn bộ 2.4 triệu bài báo đa ngành dưới dạng JSON thuần (chỉ có metadata văn bản, chưa có vector nhúng). Sau khi lọc lĩnh vực CS và ghép vector nhúng 384 chiều vào từng bản ghi, file đầu ra \texttt{arxiv\_cs\_full\_with\_vectors.jsonl} phình lên \textbf{12 GB} do mỗi bản ghi chứa thêm mảng 384 số thực dấu phẩy động.

\textbf{100k} & 100,000 & 917 MB & 860 MB \\

\textbf{Lưu ý:} Ở quy mô 100k và 500k, dữ liệu được đánh chỉ mục vào một index duy nhất (\texttt{arxiv\_bench}) chứa cả metadata văn bản lẫn vector nhúng. Ở quy mô 1.2M (Full), hệ thống sử dụng kiến trúc Dual-Index: \texttt{arxiv\_text} (1.5 GB, chỉ chứa metadata) và \texttt{arxiv\_vectors} (10.4 GB, chứa vector 384 chiều). Tổng dung lượng Dual-Index là 11.9 GB.

    \centering
    \includegraphics[width=0.6\textwidth]{index_size.png}
    \caption{Tăng trưởng dung lượng Index trên Elasticsearch theo quy mô tài liệu}
    \label{fig:index-size-chart}

\textbf{Phân tích sự tăng trưởng không gian đĩa:}
\begin{itemize}
    \item \textbf{Chỉ mục nén hiệu quả:} Kích thước index trên Elasticsearch luôn \textbf{nhỏ hơn file JSONL thô} (11.9 GB so với 12 GB ở quy mô 1.2M). Elasticsearch sử dụng các thuật toán nén cột nâng cao (LZ4 cho trường văn bản, nén mảng vector float32, Doc Values cho trường keyword) để tiết kiệm không gian lưu trữ.
    \item \textbf{Vector chiếm phần lớn dung lượng:} Ở quy mô 1.2M với kiến trúc Dual-Index, index \texttt{arxiv\_vectors} chứa vector nhúng chiếm tới \textbf{87\%} tổng dung lượng (10.4 GB / 11.9 GB). Điều này cho thấy chi phí lưu trữ chính của hệ thống Hybrid Search nằm ở cấu trúc đồ thị HNSW và các mảng vector, không phải ở Inverted Index truyền thống.
    \item \textbf{Tăng trưởng gần tuyến tính:} Dung lượng index tăng trưởng gần đúng tuyến tính theo số lượng tài liệu (100k $\rightarrow$ 860 MB, 500k $\rightarrow$ 4.3 GB, 1.2M $\rightarrow$ 11.9 GB), cho thấy hệ thống có khả năng dự đoán chi phí lưu trữ khi mở rộng quy mô.
\end{itemize}

\section{Đánh đổi giữa Chất lượng Tìm kiếm và Chi phí Tài nguyên (Trade-off Analysis)}
Việc xây dựng hệ thống tìm kiếm thực tế là một bài toán tối ưu hóa sự cân bằng giữa chất lượng tìm kiếm và chi phí tài nguyên vận hành:
\begin{enumerate}
    \item \textbf{BM25 (Chi phí thấp --- Chất lượng trung bình):} BM25 hoạt động nhẹ nhàng nhất, chỉ cần dung lượng RAM nhỏ (JVM Heap 1GB -- 2GB) và tài nguyên CPU tối thiểu. Tuy nhiên chất lượng tìm kiếm giảm sút khi gặp các truy vấn khái niệm phức tạp (NDCG giảm xuống 0.64 -- 0.67).
    \item \textbf{kNN Vector (Chi phí cao --- Chất lượng ổn định):} kNN Vector Search yêu cầu bộ nhớ RAM lớn để nạp toàn bộ đồ thị HNSW phục vụ tìm kiếm khoảng cách gần nhất. Độ trễ có xu hướng tăng cao ở lượt đầu tiên (Cold Start) nhưng cực kỳ ổn định về chất lượng khi quy mô phình to.
    \item \textbf{Hybrid Search (Chi phí cao --- Chất lượng tối ưu nhất):} Đạt điểm số NDCG@10 vượt trội hoàn toàn ($\sim$0.97 -- 0.98), mang lại kết quả chuẩn xác nhất cho mọi dạng truy vấn. Nhờ cấu hình thông minh ES Heap 6GB trên môi trường RAM 12GB WSL, Hybrid Search khắc phục được độ trễ cold-start và duy trì phản hồi mượt mà dưới 100ms (P50 = 99.0ms).
\end{enumerate}
\textbf{Kết luận:} Với ứng dụng tìm kiếm tài liệu khoa học học thuật chuyên biệt như arXiv, việc đầu tư tài nguyên phần cứng (đặc biệt là dung lượng RAM đủ lớn cho Lucene cache và cấu hình ES Heap tối ưu) để vận hành \textbf{Hybrid Search với RRF} là giải pháp tối ưu và mang lại giá trị thực tiễn lớn nhất.
